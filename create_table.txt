CREATE TABLE form (
 id INT AUTO_INCREMENT,
 name VARCHAR(30),
 second_name VARCHAR(50),
 gender VARCHAR(30),
 age TINYINT,
 partner VARCHAR(30),
 b_date DATETIME,
 family_status VARCHAR(30),
 social_status VARCHAR(30),
 live VARCHAR(30),
 weekends VARCHAR(30),
 about_forms VARCHAR(30),
 label_book VARCHAR(30),
 text_comm VARCHAR(30),
 position VARCHAR(30),
 name_field VARCHAR(30),
 mail VARCHAR(30),
 spam_cat VARCHAR(80),
 task VARCHAR(30),
 PRIMARY KEY (id)
);
